import torch
from quant_torch.analytics import tensorcontainer

def square_root_symmetric_matrix(A):
    w, v = torch.symeig(A, eigenvectors=True)
    return torch.mm(torch.mm(v, torch.diag(torch.sqrt(w[:]))), v.t())

# Implement BiCGStab. Encapsulated A, b, x so that this works generically 
def bicgstab(A,b,x,v0,rho0,epsilon=1.e-6,iter_max=100):
    # r0 = b − Ax
    # r = b - torch.mm(A,x)
    r = b - torch.mv(A,x)
    #Choose an arbitrary vector r̂0 such that (r̂0, r0) ≠ 0, e.g., r̂0 = r0 . Note that notation (x,y) applies for scalar product of vectors (x,y) = <x,y> = x·y = x ' y
    r0 = r
    #ρ0 = α = ω0 = 1
    rho = alpha = omega = rho0
    #v0 = p0 = 0
    v = p = v0

    it = 0

    finished = False
    # For i = 1, 2, 3, …
    while not finished:
        #ρi = (r̂0, ri−1)
        rho_next = torch.dot(r.T,r0)
        # β = (ρi/ρi−1)(α/ωi−1)
        beta = (rho_next/rho)*(alpha/omega)
        # pi = ri−1 + β(pi−1 − ωi−1vi−1)
        p_next = r + beta*(p - omega * v)
        # vi = Api
        v_next = torch.mv(A,p_next)
        # α = ρi/(r̂0, vi)
        alpha = rho_next / torch.dot(r0,v_next)
        #h = xi−1 + αpi
        h = x + alpha * p_next
        #If h is accurate enough, then set xi = h and quit

        #s = ri−1 − αvi
        s = r - alpha * v_next
        if s.norm() < epsilon:
            return h
        #t = As
        t = torch.mv(A,s)
        #ωi = (t, s)/(t, t)
        omega_next = torch.dot(t,s)/torch.dot(t,t)
        #xi = h + ωis
        x_next = h + omega_next * s
        #If xi is accurate enough, then quit
        #ri = s − ωit
        r_next = s - omega_next * t
        if (r_next.norm() < epsilon) or it > iter_max:
            return x_next

        # Increase iteration count
        x = x_next
        r = r_next
        p = p_next
        rho = rho_next
        omega = omega_next
        it = it + 1
