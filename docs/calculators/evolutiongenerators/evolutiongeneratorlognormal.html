import torch
import numpy as np
from quant_torch.calculators.evolutiongenerators import evolutiongeneratorbase
from quant_torch.analytics import finitedifferencegrid, matrixanalytics, tensorcontainer

class EvolutionGeneratorLognormal(evolutiongeneratorbase.EvolutionGeneratorMonteCarloBase):
    def __init__(self, data, productData, indexObservations, simulationDates,
                 forwardVarianceVector, forwardCovarianceVector):
        super(EvolutionGeneratorLognormal, self).__init__(data, productData, indexObservations)
        self.data = data
        self.productData = productData
        self.numberOfSimulations = data['NumberOfSimulations']
        self.numberOfGridPoints = data['NumberOfGridPoints']   
        self.timedelta = data['TimeDelta']
        # Level of implicitness in finite difference
        self.gamma = 1
        if 'Gamma' in data.keys():
            self.gamma = data['Gamma']
        self.forwardVarianceVector = forwardVarianceVector
        self.forwardCovarianceVector = forwardCovarianceVector
        self.simulationDates = simulationDates
        self.indexObservations = indexObservations
        self.boundaryGrid = 0
        self.dimensions = len(self.forwardCovarianceVector[0])
        self.finiteDifferenceOrthotope = finitedifferencegrid.FiniteDifferenceOrthotope(self.dimensions,self.numberOfGridPoints)
        #self.finiteDifferenceOrthotope.print()

    def createStateTensor(self):
        # Simulate this is really where most of the effort is going to be
        n = len(self.simulationDates)

        m = len(self.forwardCovarianceVector[0])
        print(m)

        # Draw random numbers
        z = torch.randn(size=(n,m,self.numberOfSimulations))

        logsamples = torch.zeros(size=(m,self.numberOfSimulations))

        sampleValues = {}

        for i,it in enumerate(self.simulationDates):
            # Need to implement the pseudosquareroot of the matrix
            dW = torch.mm(self.forwardCovarianceVector[i],z[i,:,:])
            for j in range(m):
                dW[j] = dW[j] - self.forwardVarianceVector[i][j]/2.
            logsamples = logsamples + dW

            if it in self.productData.getDatesUnderlyings().keys():
                sampleValues[it] = logsamples

        return sampleValues

    def getStateFiniteDifference(self):
        return self.finiteDifferenceOrthotope.getFiniteDifferenceGrid()

    def getOperatorMatrix(self):
        return 0


    def singleStep(self, values):
        n = self.finiteDifferenceOrthotope.getTotalDimensionality()
        #A = torch.sparse.zeros(size=(n,n))
        A = torch.ones(size=(n,n)) 
        B = torch.ones(size=(n,n)) 

        Q = self.finiteDifferenceOrthotope.getQ()

        forwardCovariance = self.forwardCovarianceVector[0]

        n = self.finiteDifferenceOrthotope.getTotalDimensionality()
        #A = torch.sparse.zeros(size=(n,n))
        A = torch.eye(n) 
        B = torch.eye(n) 

        Q = self.finiteDifferenceOrthotope.getQ()
        G = self.finiteDifferenceOrthotope.getG()        

        for i,it in enumerate(Q):
            x = Q[it].get_x()
            y = Q[it].get_y()
            diffusion = Q[it].get_diffusion()
            for jt in diffusion:
                xj = jt.get_x()
                yj = jt.get_y()                
                dx = xj - x
                j = G[jt]
                
                #A[i][j] = forwardCovariance[0]*(1./2 - 1./dx)/dx/4.
                B[i][j] = forwardCovariance[0]*(1./2 + 1./dx)/dx/4.
                #A[i][i] = A[i][i] + forwardCovariance[0]/dx/dx/4.
                B[i][i] = B[i][i] - forwardCovariance[0]/dx/dx/4.

        #print(A)
        #print(B)

        #A = A.to_sparse()

        At = tensorcontainer.ContainerTensor(A)

        v0 = torch.zeros(n)
        #v0 = tensorcontainer.ContainerTensor(v0)
        rho0 = torch.tensor(1.0)
        #rho0 = tensorcontainer.ContainerTensor(rho0)

        for i,it in enumerate(values):
            y = values[i]
            y = torch.mv(B,y)

            yt = tensorcontainer.ContainerTensor(y)
            z = matrixanalytics.bicgstab(A,y,y,v0,rho0,1.e-4)

            values[i] = z

        # No step yet
        values = values


    def getFiniteDifferenceDates(self):
        return self.simulationDates


    def getValue(self, date, index, stateTensor):
        return self.indexObservations[index][date].getValue(date,stateTensor)
